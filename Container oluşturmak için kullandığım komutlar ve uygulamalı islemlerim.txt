:: MACOS DOCKER APPLİED TERMİNAL COMMANDS :: 
:: IMAGE LİSTELE: 
- muratcelik@Murat-MacBook-Pro ~ % docker images
docker images
docker image ls

::IMAGE SİL: 
docker container rm CONTAINERID
docker rmi -f postgres:latest
docker image rm *  //tüm imageları sil.


:: CONTAİNER LİSTELE
- Çalışan containerlar için
docker ps

- Çalışan ve stop olmuş tüm containerlar için
docker ps -a

:: CONTAİNER SİL
docker container m -f postgres

:: CONTAİNER ID
-Detaylı container ID bakma
docker container ls -a --no-trunc

Sadece  container ID bakma;
docker container ls -a -q

:: CONTAİNER ÇALIŞTIR;
docker container start CONTAINERID
docker container start -a CONTAINERID

:: CONTAİNER DURDUR;
docker container stop CONTAINERID

- Kill 10 sn shutdown eder.
docker container kill CONTAINERID
docker container prune (stop edlmiş tüm containerları sil)

:: CONTAİNER BİLGİSİ 
- Ayrıntılı Bilgi;
docker container inspect CONTAINERID

- Contains (IPAdress geçenleri getir)
docker container inspect CONTAINERID | grep IPAdress

:: PORT MAPPİNG
- Port eklemek için;
docker container run -d -p 5000:80 nginx 

- Hangi port açık;
docker container port CONTAINERID

:: PLUGİN
docker plugin ls

- Plugin disable-enable;
docker plugin disable PLUGINNAME
docker plugin enable PLUGINNAME

- Plugin ayrıntısını görme;
docker plugin inspect PLUGINNAME

- Plugin Kaldır;
docker plugin rm PLUGINNAME


:: VOLUME
docker info
 Docker Root Dir: /var/lib/docker   (docker volume örnek adresi )

- Docker volume oluştur;
docker volume create --name VOLUMENAME

:: HELP
docker container --help
docker image --help
docker tag   --help
docker tag SOURCE_IMAGE[:TAG] TARGET_IMAGE[:TAG]
